#Ce code permet de modeliser l'experience de Stroop : celle-ci consiste a presenter differents mots de couleurs differentes

import random
from expyriment import design, control, stimuli

WHITE = (255,255,255)
RED = (255,0,0)
YELLOW = (255,255,0)
GREEN = (0,255,0)
BLUE = (0,0,255)

N_TRIALS = 40
MIN_WAIT_TIME = 1000
MAX_WAIT_TIME = 2000
MAX_RESPONSE_DELAY = 10000

exp = design.Experiment(name="Visual Detection", text_size=40)
control.set_develop_mode(on=True)
control.initialize(exp)

WORDSLIST = ["YELLOW", "RED", "BLUE", "GREEN"]
WORDSLIST = WORDSLIST * 5
COLORLIST = [YELLOW,RED,BLUE,GREEN]
COLORLIST5 = COLORLIST * 5

COHERENTLIST = []
for i in range (20):
    i = stimuli.TextLine(WORDSLIST[i],position=(0,0),text_size=50,text_colour=COLORLIST5[i])
    COHERENTLIST.append(i)

INCOHERENTLIST = []
for k in range (20) :
    a = random.randint(0,19)
    if a%4 == k%4 :
        a = a + 1
    k = stimuli.TextLine(WORDSLIST[k],position=(0,0),text_size=50,text_colour=COLORLIST5[a])
    INCOHERENTLIST.append(k)

TARGETLIST = COHERENTLIST + INCOHERENTLIST
random.shuffle(TARGETLIST)

blankscreen = stimuli.BlankScreen()

instructions = stimuli.TextScreen("Instructions",
    f'''From time to time, a word will appear in front of you. 

    Your task is to press the SPACEBAR as quickly as possible when you see it (We measure your reaction-time).

    There will be {N_TRIALS} trials in total.

    Press the spacebar to start.)''')

#exp.add_data_variable_names(['trial', 'wait', 'respkey', 'RT'])

for i_trial in range(N_TRIALS):
    blankscreen.present()
    waiting_time = random.randint(MIN_WAIT_TIME, MAX_WAIT_TIME)
    exp.clock.wait(waiting_time)
    TARGETLIST[i_trial].present()
    key, rt = exp.keyboard.wait(duration=MAX_RESPONSE_DELAY)
    #exp.data.add([i_trial, waiting_time, key, rt])

control.end()
